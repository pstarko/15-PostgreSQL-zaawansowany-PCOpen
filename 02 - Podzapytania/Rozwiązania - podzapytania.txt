/*
1. Wyświetl nazwiska i stanowiska pracowników pracujących w tym samym dziale co pracownik o nazwisku Miller (5 wierszy)
*/

select
	e.first_name ,
	e.last_name ,
	j.job_title 
	from employees e join jobs j on e.job_id  = j.job_id 
where
	e.department_id = (select department_id from employees
						where last_name = 'Miller')



/*
2. Wyświetl następujące dane o najdłużej zatrudnionym pracowniku na stanowisku FI_ACCOUNT (1 wiersz)
*/

select
	e.employee_id ,
	e.first_name ,
	e.last_name ,
	e.job_id ,
	e.hire_date
	from employees e 
where
	e.hire_date = (select min(hire_date) from employees
					where job_id = 'FI_ACCOUNT')
	and e.job_id = 'FI_ACCOUNT'			
					
/*
3. Zobacz najdłużej pracujących pracowników każdego działu. (12 wierszy)
*/

select 
	e.last_name ,
	e.hire_date ,
	d.department_id ,
	d.department_name 
	from employees e join departments d on e.department_id = d.department_id 
where 
	(e.department_id, hire_Date) in (
									select 
										department_id,
										min(hire_Date)
										from employees e 
									group by department_id )




/*
4.
Wyświetlaj te nazwy stanowisk, w których nie zatrudniono nikogo z działu MARKETING. Wykorzystaj podzapytanie w podzapytaniu (17 wierszy)
*/

select * from jobs
where
	job_id in (select job_id from employees
					where department_id not in (select department_id
												from departments
												where 
												department_name='Marketing'))
												
												
select *
from jobs j
where not exists (
				select 1
					from employees e
					where e.job_id = j.job_id
					and e.department_id = (
							select d.department_id
							from departments d
							where d.department_name = 'Marketing'));
/*
7.
Użyj skorelowanego podzapytania, aby wyświetlić 
informacje o pracowniku niezatrudnionym w żadnym dziale (użyj właściwej formy operatora EXISTS) (1 wiersz)
*/

select
    e.first_name,
    e.last_name
    from employees e 
    where not exists (select 1 from departments dd where dd.department_id = e.department_id)
												
												
/*
8. Wyświetl nazwę działu, który płaci Twoim pracownikom najwięcej pieniędzy miesięcznie(1 wiersz)
*/
select
	d.department_name
	from employees e join departments d on e.department_id = d.department_id 
group by 
	d.department_name 
having
	sum(e.salary) = (
					select
						max(t1.suma)
						from (
							select
								department_id,
								sum(salary) suma
								from employees
							group by 
								department_id ) t1 )
