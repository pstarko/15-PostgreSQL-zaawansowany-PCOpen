WITH RECURSIVE hierarchia AS (
-- Kotwica: pracownicy bez managera (top-level)
	SELECT
		employee_id,
		last_name,
		manager_id,
		1 AS poziom,
		last_name::TEXT AS sciezka
	FROM employees
		WHERE manager_id IS NULL
UNION ALL
-- Rekurencja: dołączamy podwładnych
	SELECT
		e.employee_id,
		e.last_name,
		e.manager_id,
		h.poziom + 1,
		h.sciezka || ' > ' || e.last_name
	FROM employees e JOIN hierarchia h ON e.manager_id = h.employee_id)
SELECT
*
FROM hierarchia
ORDER BY sciezka;
