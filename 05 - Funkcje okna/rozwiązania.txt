/*
1. Pokaż jak kształtowała się liczba zatrudnien w poszczególnych latach. 
W dodatkowej kolumnie dodaj narastające wartości ilości zatrudnień w sumie dla każdego roku (8 wierszy)
*/

select
	date_part('year', hire_date),
	count(*) ilosc_zatrudnien,
	sum(count(*)) over (order by date_part('year', hire_date)) narast_
	from employees
group by
	date_part('year', hire_date)

/*
2. Zmień powyższy raport o informacje na jaki etat zostali zatrudnieni pracownicy. 
Również dodaj kolumnę z wartościami narastającymi(47 wierszy)
*/
	
	select
	date_part('year', hire_date),
	j.job_title ,
	count(*) ilosc_zatrudnien,
	sum(count(*)) over (order by date_part('year', hire_date), j.job_title ) narast_
	from employees e join jobs j on e.job_id = j.job_id 
group by
	date_part('year', hire_date),
	j.job_title
