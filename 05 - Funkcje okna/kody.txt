SELECT
	last_name,
	hire_date,
	LAG(last_name, 1,'----') OVER (ORDER BY hire_date) poprzednik,
	LEAD(last_name,1,'----') OVER (ORDER BY hire_date) nastepny
FROM 
	employees

SELECT
	last_name,
	hire_date,
	department_id,
	LEAD(last_name,1) OVER (partition by department_id ORDER BY hire_date) AS nastepny
FROM employees


SELECT
	department_id,
	STRING_AGG(last_name,',') nazwiska
from employees
group by department_id


SELECT
	department_id,
	STRING_AGG(last_name,',' order by last_name) nazwiska
from employees
group by department_id

SELECT
	distinct department_id,
	FIRST_VALUE(last_name) over (partition by department_id order by salary DESC) last_name
from employees

SELECT
	distinct department_id,
	LAST_VALUE(last_name) over (partition by department_id order by salary DESC) last_name
from employees

select
	department_id,
	last_name,
	salary
from employees
where department_id in (10,20,30)


SELECT
	distinct department_id,
	LAST_VALUE(last_name) over (partition by department_id order by salary desc
										rows between unbounded preceding
											 and unbounded following) last_name
from employees


select 
	distinct department_id,
	sum(salary) over (partition by department_id) suma
from employees


select
	hire_date,
	salary,
	sum(salary) over (order by hire_date
					ROWS BETWEEN 1 PRECEDING 
					       	 AND 1 FOLLOWING) grow
from employees


select
	hire_date,
	salary,
	sum(salary) over (order by hire_date
					ROWS BETWEEN 1 PRECEDING 
					       	 AND current row) grow
from employees

select
	hire_date,
	salary,
	sum(salary) over (order by hire_date
					RANGE BETWEEN unbounded PRECEDING 
					       	 AND current row) _range,
	sum(salary) over (order by hire_date
					ROWS BETWEEN unbounded PRECEDING 
					       	 AND current row) _rows
from employees


select
	hire_date,
	salary,
	sum(salary) over (order by hire_date
					ROWS BETWEEN 2 PRECEDING 
					       	 AND 1 PRECEDING) grow
from employees


select
	hire_date,
	salary,
	sum(salary) over (order by hire_date	
				ROWS BETWEEN UNBOUNDED PRECEDING 
								AND CURRENT ROW) suma_kroczaca
from employees

select
	department_id,
	sum(salary),
	sum(sum(salary)) over (order by department_id
							ROWS BETWEEN UNBOUNDED PRECEDING 
									 AND CURRENT ROW) okno
from employees
group by
	department_id
