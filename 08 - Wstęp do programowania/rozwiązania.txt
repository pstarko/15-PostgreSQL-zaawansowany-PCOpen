/*
1. Użytkownik podaje dwie liczby X i Y. 
Następnie pokazujesz mu liczby całkowite od X do Y.
*/

DO $$
DECLARE
    x INT := 15;  -- tutaj podaj pierwszą liczbę
    y INT := 10; -- tutaj podaj drugą liczbę
    i INT;
    temp INT;
BEGIN
    -- Zamiana miejscami jeśli x > y
    IF x > y THEN
        temp := x;
        x := y;
        y := temp;
    END IF;

    -- Wyświetlenie liczb od x do y
    FOR i IN x..y LOOP
        RAISE NOTICE 'Liczba: %', i;
    END LOOP;
END $$;


/*
2.
Stwórz pętlę pokazującą informację: Liczba 50, Liczba 49…., Liczba 1. 
Pętla rozpoczyna się od 50 a kończy na 1. Pomiń liczby parzyste pomiędzy 20 a 30.
*/

DO $$
DECLARE
    i INTEGER;
BEGIN
    FOR i IN REVERSE 50..1 LOOP
        -- Pomijamy liczby parzyste między 20 a 30
        IF NOT (i BETWEEN 20 AND 30 AND i % 2 = 0) THEN
            RAISE NOTICE 'Liczba %', i;
        END IF;
    END LOOP;
END $$;

/*
Napisz skrypt, który:
•
Dla każdego klienta zaktualizuje kolumnę Status w zależności od wartości Saldo:
	o VIP jeśli Saldo > 10000
	o Standard jeśli Saldo BETWEEN 1000 AND 10000
	o Zagrożony jeśli Saldo < 1000
	o Zmień dane w tabeli i wypisz je w konsoli
	
Użyj: CASE
*/


DO $$
DECLARE
    klient_rec RECORD;
    kursor CURSOR FOR SELECT * FROM Klienci;
BEGIN
    -- Aktualizacja statusu dla wszystkich klientów
    UPDATE Klienci
    SET Status = CASE
        WHEN Saldo > 10000 THEN 'VIP'
        WHEN Saldo BETWEEN 1000 AND 10000 THEN 'Standard'
        WHEN Saldo < 1000 THEN 'Zagrożony'
    END;
    
    -- Wypisanie zaktualizowanych danych w konsoli
    OPEN kursor;
    LOOP
        FETCH kursor INTO klient_rec;
        EXIT WHEN NOT FOUND;
        
        RAISE NOTICE 'Klient ID: %, Imię: %, Nazwisko: %, Saldo: %, Status: %',
            klient_rec.Id, klient_rec.Imie, klient_rec.Nazwisko, 
            klient_rec.Saldo, klient_rec.Status;
    END LOOP;
    CLOSE kursor;
END $$;



/*
5.
Napisz program do obliczania kolejnych liczb całkowitych. 
Np. po podaniu liczby 5 otrzymasz 15(czyli 1+2+3+4+5)

*/

DO $$
DECLARE
    liczba INT := 6; -- Podaj tutaj liczbę, dla której chcesz obliczyć sumę
    suma INT := 0;
    i INT;
BEGIN
    -- Obliczanie sumy za pomocą pętli
    FOR i IN 1..liczba LOOP
        suma := suma + i;
    END LOOP;
    
    -- Wyświetlenie wyniku
    RAISE NOTICE 'Suma liczb od 1 do % wynosi %', liczba, suma;
    
    -- Alternatywnie, można użyć wzoru na sumę ciągu arytmetycznego: n*(n+1)/2
    RAISE NOTICE 'Suma obliczona wzorem: %', (liczba * (liczba + 1)) / 2;
END $$;



/*
6.
Napisz program, który zwróci Ci liczbę liter i cyfr w podanych ciągu znaków i cyfr. 
Znajdź wcześniej odpowiednią funkcję, która Ci to umożliwi. 
Do sprawdzenia czy znak jest liczbą czy literą wykorzystaj strukturę: 
IF znak ~ ’[0-9]’ THEN …

*/

DO $$
DECLARE
    tekst TEXT := 'Agent8007'; -- Podaj tutaj swój ciąg znaków
    liczba_liter INT := 0;
    liczba_cyfr INT := 0;
    i INT;
    znak CHAR;
BEGIN
    -- Iteracja przez każdy znak w ciągu
    FOR i IN 1..LENGTH(tekst) LOOP
        znak := SUBSTRING(tekst,i,1);
        
        -- Sprawdzenie czy znak jest literą
        IF znak ~ '[A-Za-z]' THEN
            liczba_liter := liczba_liter + 1;
        -- Sprawdzenie czy znak jest cyfrą
        ELSIF znak ~ '[0-9]' THEN
            liczba_cyfr := liczba_cyfr + 1;
        END IF;
    END LOOP;
    
    -- Wyświetlenie wyników
    RAISE NOTICE 'Ciąg znaków: %', tekst;
    RAISE NOTICE 'Liczba liter: %', liczba_liter;
    RAISE NOTICE 'Liczba cyfr: %', liczba_cyfr;
END $$;
