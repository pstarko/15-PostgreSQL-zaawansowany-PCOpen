/*
1. Pokaż imię, nazwisko, numer działu i stanowisko dla wszystkich pracowników. 
Nie używaj konstrukcji „where department_id is null”, rozwiąż problem połączeniami tabel (106 wierszy)
*/
select
	e.first_name ,
	e.last_name ,
	e.department_id ,
	j.job_title 
	from employees e join jobs j on e.job_id  = j.job_id 
					 join departments d on e.department_id = d.department_id 


/*
2. Pokaż pracowników pracujących w Seattle i Londynu (19 wierszy)
*/
					 
select
	e.first_name ,
	e.last_name ,
	l.city 
	from employees e join departments d on e.department_id = d.department_id 
					 join locations l on d.location_id = l.location_id 
where
	l.city in ('Seattle', 'London')


/*
3. Pokaż poniższy raport nazwisk i nazw departamentów. 
Uwzględnij departamenty, które nie zatrudniają pracowników. Nie bierz pod uwagę zespołu Operations(121 wiersze)
*/
select
	e.last_name ,
	d.department_name 
	from employees e  right join departments d on e.department_id = d.department_id 
where
	d.department_name not in ('Operations')

/*
4. Pokaż ilość zatrudnionych pracowników w Stretford, Bombay, Sydney, London, Munich i Seattle(6 wierszy)
*/
					 
select
	l.city,
	count(e.employee_id)
	from employees e join departments d on e.department_id = d.department_id 
					 right join locations l on d.location_id = l.location_id 
where
	l.city in ('Seattle', 'London','Munich','Sydney','Bombay','Stretford')
group by
	l.city 

/*
5. Pokaż liczbę pracowników, którzy pracują w prowincji stanowej(State Province), zaczynającej się od litery T (1 wiersz) (1 wiersz)
*/
select
	count(e.employee_id)
	from employees e join departments d on e.department_id = d.department_id 
					 right join locations l on d.location_id = l.location_id 
where
	l.state_province like 'T%'

/*
6. Pokaż poniższe informacje o pracownikach, którzy są zatrudnieni na stanowisku Accountant oraz Purchasing Clerk, 
których roczny dochód przekracza 37200. 
Roczny dochód jest dwunastokrotnością wynagrodzenia podstawowego wraz z prowizją (5 wierszy)
*/
					 
select
	e.last_name ,
	j.job_title ,
	(e.salary + (salary * coalesce(e.commission_pct,0))) * 12 salary
	from employees e join jobs j on e.job_id =j.job_id 
where
	j.job_title in ('Accountant', 'Purchasing Clerk')
	and (e.salary + (salary * coalesce(e.commission_pct,0))) * 12 > 37200


/*
7. Pokaż nazwy departamentów wraz z nazwiskami ich szefów i miejscem pracy (11 wierszy)

**/

select
	e.first_name as boss_first_name,
	e.last_name as boss_last_name,
	d.department_name,
	l.city
from departments d join employees e on d.manager_id = e.employee_id
					join locations l on d.location_id = l.location_id



/*
8. Zmodyfikuj powyższe zapytanie w taki sposób, by zostały wyświetlone departamenty nie posiadające szefa. 
Wybierz właściwy typ połączenia tabel(27 wierszy)
*/
					 
select
e.first_name as boss_first_name,
e.last_name as boss_last_name,
d.department_name,
l.city
from departments d left join employees e on d.manager_id = e.employee_id
       			   join locations l on d.location_id = l.location_id
