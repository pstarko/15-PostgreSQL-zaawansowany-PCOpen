--Partycjonowanie tabel w PostgreSQL

/*
Partycjonowanie to technika podziału dużej tabeli na mniejsze, fizycznie oddzielne fragmenty (partycje), 
które zachowują się jak jedna logiczna tabela. 
Świetnie sprawdza się przy tabelach z milionami/miliardami wierszy.

Kiedy warto partycjonować?
	Tabele z danymi historycznymi (logi, zdarzenia, transakcje)
	Zapytania filtrujące po zakresie dat lub kategorii
	Potrzeba szybkiego usuwania starych danych (DROP PARTITION zamiast DELETE)
	Tabele > kilkaset GB
*/

-- Krok 1 — Partycjonowanie RANGE (po dacie)

	drop table orders cascade;
		
		CREATE TABLE orders (
			id          BIGSERIAL,
			created_at  TIMESTAMPTZ NOT NULL,
			customer_id INT,
			amount      NUMERIC(10,2),
			status      TEXT
		) PARTITION BY RANGE (created_at);

		-- Partycje roczne
		CREATE TABLE orders_2023
			PARTITION OF orders
			FOR VALUES FROM ('2023-01-01') TO ('2024-01-01');

		CREATE TABLE orders_2024
			PARTITION OF orders
			FOR VALUES FROM ('2024-01-01') TO ('2025-01-01');

		CREATE TABLE orders_2025
			PARTITION OF orders
			FOR VALUES FROM ('2025-01-01') TO ('2026-01-01');

		-- Partycja domyślna (dla wartości spoza zakresów)
		CREATE TABLE orders_default
			PARTITION OF orders DEFAULT;

--1. Partition pruning — Postgres skanuje tylko orders_2024:		
EXPLAIN ANALYZE
SELECT * FROM orders
WHERE created_at BETWEEN '2024-01-01' AND '2024-12-31';

--2. Sprawdzenie, w której partycji fizycznie leżą dane:
SELECT tableoid::regclass AS partition, id, created_at, amount, status
FROM orders
ORDER BY created_at;
		

--3. Agregacja per rok — każda partycja liczona osobno:
explain
SELECT
  DATE_TRUNC('year', created_at) AS year,
  COUNT(*)                        AS orders_count,
  SUM(amount)                     AS total_revenue,
  AVG(amount)                     AS avg_order
FROM orders
GROUP BY 1
ORDER BY 1;

---> Partycjonowanie LIST (po kategorii)

drop table products cascade;
		
		CREATE TABLE products (
			id       BIGSERIAL,
			name     TEXT,
			category TEXT NOT NULL,
			price    NUMERIC(10,2)
		) PARTITION BY LIST (category);

		CREATE TABLE products_electronics
			PARTITION OF products
			FOR VALUES IN ('electronics', 'computers', 'phones');

		CREATE TABLE products_clothing
			PARTITION OF products
			FOR VALUES IN ('clothing', 'shoes', 'accessories');

		CREATE TABLE products_other
			PARTITION OF products DEFAULT;

EXPLAIN ANALYZE
SELECT * FROM products
WHERE category IN ('electronics', 'phones');

SELECT tableoid::regclass AS partition, id, name, category, price
FROM products
ORDER BY partition, category;	


SELECT DISTINCT ON (tableoid::regclass)
  tableoid::regclass AS partition,
  name,
  category,
  price
FROM products
ORDER BY tableoid::regclass, price DESC;

    ATTACH PARTITION products_electronics

-- 1. Odepnij partycję
ALTER TABLE products DETACH PARTITION products_electronics;

ALTER TABLE products
    ATTACH PARTITION products_electronics
    FOR VALUES IN ('electronics', 'computers', 'phones', 'tablets');

CREATE TABLE products_tablets
    PARTITION OF products
    FOR VALUES IN ('tablets');

--Porównanie RANGE vs LIST
Cecha				RANGE (orders)					LIST (products)
Klucz podziału		Zakres ciągły (daty, liczby)		Dyskretne wartości (kategorie, statusy)
Pruning				WHERE created_at > ...			WHERE category = ...
Typowy use case		Dane czasowe, logi				Słowniki, enumy, regiony
Partycja DEFAULT		Dane spoza zakresów				Nieznane/nowe kategorie		


--> Partycjonowanie HASH (równomierne rozłożenie)

		CREATE TABLE events (
			id      BIGSERIAL,
			user_id INT NOT NULL,
			payload JSONB
		) PARTITION BY HASH (user_id);

		-- 4 partycje — modulo 4
		CREATE TABLE events_p0 PARTITION OF events FOR VALUES WITH (MODULUS 4, REMAINDER 0);
		CREATE TABLE events_p1 PARTITION OF events FOR VALUES WITH (MODULUS 4, REMAINDER 1);
		CREATE TABLE events_p2 PARTITION OF events FOR VALUES WITH (MODULUS 4, REMAINDER 2);
		CREATE TABLE events_p3 PARTITION OF events FOR VALUES WITH (MODULUS 4, REMAINDER 3);
		
--> Indeksy na partycjach

		-- Indeks na tabeli nadrzędnej → automatycznie tworzony na wszystkich partycjach
		CREATE INDEX ON events (id);
		CREATE INDEX ON events (user_id);

		-- Lub indeks tylko na konkretnej partycji
		CREATE INDEX ON events_p0 (id);
		
--> Wstawianie i sprawdzanie danych

		-- Wstaw dane testowe
		INSERT INTO events (user_id, payload)
		SELECT
		    (random() * 99999 + 1)::INT AS user_id,
		    jsonb_build_object(
		        'event_type', (ARRAY['click', 'view', 'purchase', 'login', 'logout'])[floor(random() * 5 + 1)::INT],
		        'timestamp',  now() - (random() * interval '365 days'),
		        'session_id', md5(random()::TEXT),
		        'metadata', jsonb_build_object(
		            'ip',      concat(
		                           (random()*255)::INT, '.',
		                           (random()*255)::INT, '.',
		                           (random()*255)::INT, '.',
		                           (random()*255)::INT
		                       ),
		            'browser', (ARRAY['Chrome', 'Firefox', 'Safari', 'Edge'])[floor(random() * 4 + 1)::INT],
		            'os',      (ARRAY['Windows', 'macOS', 'Linux', 'Android', 'iOS'])[floor(random() * 5 + 1)::INT]
		        )
		    ) AS payload
		FROM generate_series(1, 100000);

		-- Sprawdź rozkład danych po partycjach
		SELECT
			tableoid::regclass AS partycja,
			COUNT(*) AS liczba_wierszy
		FROM events
		GROUP BY tableoid;
